	spawn_elizabeth_bathory = yes

}
#1614
spawn_elizabeth_bathory = {
	if = {
		limit = { 
			year = 1614
			NOT = { year > 1614 }
		}
		random_province = {
			limit = {
				culture = "hungarian"
			}
			create_character = {
				name = "Elizabeth Bathory"
				age = 24
				dna="bgkff0jaca0"
				properties="mc0j0k0000000000000"
				trait = mortal
				trait = canon
				trait = wyrm-tainted
				trait = great_person
				trait = grey_eminence
				trait = impaler
				trait = proud
				trait = cruel
				trait = diligent
				trait = schemer
				trait = deceitful
				female = yes
				religion = innocent
				dynasty = none
				culture = "hungarian"
				attributes = {
					diplomacy = 12
					intrigue = 15
					learning = 13
				}
			}
			new_character = {
				save_event_target_as = great_person
				set_character_flag = great_person
				set_character_flag = elizabeth_bathory
				set_character_flag = great_tyrant
				pod_spawn_great_person_effect = yes
			}
		}
	}
}
