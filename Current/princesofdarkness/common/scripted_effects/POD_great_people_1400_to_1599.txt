#1497
spawn_imperia_cognati = {
	if = {
		limit = { 
			year = 1497
			NOT = { year > 1497 }
		}
		random_province = {
			limit = {
				culture = "frankish"
			}
			create_character = {
				name = "Imperia Cognati"
				age = 16
				dna="cebaggbdaap"
				properties="0i0cad"
				trait = mortal
				trait = canon
				trait = great_person
				trait = grey_eminence
				trait = fair
				trait = quick
				trait = seductress
				trait = lustful
				trait = proud
				trait = ambitious
				trait = gregarious
				trait = greedy
				female = yes
				religion = innocent
				dynasty = none
				culture = "italian"
				attributes = {
					diplomacy = 15
					intrigue = 12
				}
			}
			new_character = {
				save_event_target_as = great_person
				set_character_flag = great_person
				set_character_flag = imperia_cognati
				set_character_flag = great_seducer
				pod_spawn_great_person_effect = yes
			}
		}
	}
}

		limit = { 
			year = 1415
			NOT = { year > 1415 }
		}
		random_province = {
			limit = {
				culture = "italian"
			}
			create_character = {
				name = "Leonardo da Vinci"
				age = 35
				dna="bce0c0aaba0"
				properties="lc0dfe000000000000"
				trait = mortal
				trait = canon
				trait = great_person
				trait = mastermind_theologian
				trait = genius
				trait = administrator
				trait = architect
				trait = scholar
				trait = theologian
				trait = diligent
				female = no
				religion = innocent
				dynasty = none
				culture = "italian"
				attributes = {
					learning = 12
					diplomacy = 14
					stewardship = 12
				}
			}
			new_character = {
				save_event_target_as = great_person
				set_character_flag = great_person
				set_character_flag = leonardo_da_vinci
				set_character_flag = great_artist
				pod_spawn_great_person_effect = yes
			}
		}
	}
}
