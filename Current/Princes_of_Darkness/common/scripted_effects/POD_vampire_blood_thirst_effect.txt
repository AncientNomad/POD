
POD_vampire_blood_thirst_increase_effect = {

FROM = {

if = { limit = { has_character_modifier = dire_blood_thirst	}
			FROM = {
					character_event = { id = POD_VAMPIRE_BLOOD_THIRST.5 }			
				}
	}	
	
if = { limit = { has_character_modifier = dreadful_blood_thirst	}
			FROM = {
					character_event = { id = POD_VAMPIRE_BLOOD_THIRST.4 }			
				}
	}	
	
if = { limit = { has_character_modifier = severe_blood_thirst	}
			FROM = {
					character_event = { id = POD_VAMPIRE_BLOOD_THIRST.3 }			
				}
	}	
	
if = { limit = { has_character_modifier = blood_thirst }
			FROM = {
					character_event = { id = POD_VAMPIRE_BLOOD_THIRST.2 }			
				}
	}		
	
if = { limit = { 
				NOT = { has_character_modifier = dire_blood_thirst }
				NOT = { has_character_modifier = dreadful_blood_thirst }
				NOT = { has_character_modifier = severe_blood_thirst }
				NOT = { has_character_modifier = blood_thirst }		
			}
			FROM = {
					character_event = { id = POD_VAMPIRE_BLOOD_THIRST.1 }			
				}
			}

		}	

}

	}
	}

POD_vampire_blood_thirst_dire_effect = {
		
# Diabolists are twice as likely to go into torpor (20%) or become a wight (10%) due to dire blood thirst
		
		if = { limit = { trait = diabolist }
			random_list = {
					10 = {  FROM = { character_event = { id = POD_VAMPIRE_WIGHT.1 } } }
					10 = {  FROM = { character_event = { id = POD_VAMPIRE_TORPOR.2 } } }
		
				}	
			}

		if = { limit = { NOT = { trait = diabolist } }
			random_list = {
					5 = { FROM = { character_event = { id = POD_VAMPIRE_WIGHT.1 } } }
					5 = { FROM = { character_event = { id = POD_VAMPIRE_TORPOR.2 } } }
				}
			}
	}
