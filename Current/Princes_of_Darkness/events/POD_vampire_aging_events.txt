character_event = {
    id = vampire_aging_events.8					
    hide_window = yes
    is_triggered_only = yes
    
    min_age = 100
    trigger = {
        trait = torpor
		trait = vampire
		NOT = { trait = canon }
		NOT = { trait = methuselah }
		NOT = { trait = antediluvian } 
    }

    immediate = {  
	
		if = { limit = { 
				trait = torpor
				trait = vampire
				NOT = { has_living_children = yes }
				NOT = { has_lover = yes }
				NOT = { is_married = yes }
				NOT = { num_of_claims = 1 }
				#NOT = { num_of_rivals = 1 }
				#NOT = { num_of_friends = 1 }
				}		
			random = {
					chance = 50
					death = "yes"
					}
			}
		
		}
    }
