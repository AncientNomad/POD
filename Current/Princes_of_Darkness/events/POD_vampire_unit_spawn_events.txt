#Tremere AI Spawns gargoyles bi-annually by chance
character_event = {
	id = POD_VAMPIRE_UNIT_SPAWN.6
	hide_window = yes
	is_triggered_only = yes
	only_playable = yes
	ai = yes
	min_age  = 50
	only_capable = yes
	
	immediate = {
		if = { 
			limit = {
				trait = vampire
				trait = tremere
				trait = thaumaturgy_5
			}
		random = {
			chance = 40
			location = {
			ROOT = {
				spawn_unit = {
					province = PREV
					troops = {
						mass_gargoyles = {
							50 50
						}
						mass_gargoyles = {
							50 50
						}
						mass_gargoyles = {
							50 50
						}
						mass_gargoyles = {
							50 50
						}						
					}
				
				}
			}
			}
		}
		}
	}
}
