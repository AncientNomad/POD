# revenant events

namespace = revenant_events

character_event = {
	id = revenant_events.1
	#desc = "POD_REVENANT_CHILDREN_DESC"
	
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		trait = mortal
		NOT = { trait = revenant }
		NOT = { trait = ghoul }
		OR =  {
			mother = { trait = revenant }
			father = { trait = revenant } 
			father = { trait = ghoul }
			mother = { trait = ghoul } 
		}
	}

	#both parents ghouls is 65% chance, both parents revenants is 75% chance
	immediate = {
		random_list = { 
			50 = {
			}
			50 = {
				modifier = {
					factor = 1.4
					mother = { trait = ghoul }
				}
				modifier = {
					factor = 1.4
					father = { trait = ghoul }
				}
				modifier = {
					factor = 1.7
					mother = { trait = revenant }
				}
				modifier = {
					factor = 1.7
					father = { trait = revenant }
				}
				character_event = { id = revenant_events.3 days = 3 }
			}
		}
	}

}

character_event = {

	#desc = "POD_REVENANT_DEATH_DESC"
	
	hide_window = yes
	min_age = 220
	
	trigger = {
		OR = {
			trait = revenant

			}	
 		}
			
	mean_time_to_happen = {

	modifier = {
			factor = 1.5
			trait = ghoul 
			}	
	
	modifier = {
			factor = 0.5
			trait = wolf-blooded
			}
		}
	
	immediate = {
			death = "yes"
		}
}

character_event = {
	id = revenant_events.3
	#desc = "POD_REVENANT_CHILDREN_DESC"
	
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		NOT = { trait = revenant }
	}
	
	immediate = {

	}

}
