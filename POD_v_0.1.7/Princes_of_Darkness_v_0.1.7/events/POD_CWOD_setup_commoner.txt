	#desc = "POD_SETUP_COMMONER_DESC"
	picture = GFX_evt_comet

		NOT = { trait = mage }
		NOT = { trait = mummy }
		NOT = { trait = changeling }
		NOT = { trait = promethean }

		NOT = { trait = demon }
		NOT = { trait = revenant }
		NOT = { trait = storyteller }
		OR = {
			has_global_flag = bookmark1230
			has_global_flag = bookmark1500
		}

	mean_time_to_happen = {
		days = 1
		modifier = {
		factor = 0.1

	immediate = {

		hidden_tooltip = {	

		# children
		if = {
							limit = {
								age = 1
								NOT = { age = 16 }
								NOT = { is_adult = yes }
								NOT = { is_ruler = yes }
								NOT = { trait = mortal }
								NOT = { trait = vampire }
								NOT = { trait = werewolf }

		NOT = { trait = werewolf }
		NOT = { trait = mage }
		NOT = { trait = mummy }

		NOT = { trait = geist }
		NOT = { trait = demon }
		NOT = { trait = revenant }		
		NOT = { trait = storyteller }

# vamp mercs

										NOT = { religion = revenant_inconnu }
										NOT = { religion = revenant_giovanni }

										NOT = { religion = revenant_mayaparisataya }

											religion_group = masquerade_camarilla_group
											religion_group = sabbat_group
											religion_group = inconnu_group
											religion_group = masquerade_independent_group
											religion = laibon
										}
										OR = {
											liege = { mercenary = yes }
											liege = { holy_order = yes }
											trait = peasant_leader
										}
									
								}
						add_trait = vampire
						character_event = { id = CWOD_vampire_commoner.1 }
					}

					if = { limit = {	
									OR = {
										religion = revenant_high_clans 
										religion = revenant_ashirra 
										religion = revenant_anarchs 
										religion = revenant_autarkis

										religion = revenant_mayaparisataya

										}

											liege = { mercenary = yes }

						character_event = { id = CWOD_werewolf_commoner.1 }

					91 = { 
						# mortal
						add_trait = mortal
						character_event = { id = mortal_commoner.1 }

						add_trait = vampire
						character_event = { id = CWOD_vampire_commoner.1 }

					1 = {

						add_trait = werewolf
						add_trait = hishu
						character_event = { id = CWOD_werewolf_commoner.1 }

					1 = {
						# mage
						add_trait = mage
						character_event = { id = mage_commoner.1 }
						}
					1 = {
						# changeling
						add_trait = changeling
						character_event = { id = changeling_commoner.1 }
						}
					1 = {
						# mummy
						add_trait = mummy

						character_event = { id = promethean_commoner.1 }

						}
						
					
					
					}		
				}			
			}	
	
		}
